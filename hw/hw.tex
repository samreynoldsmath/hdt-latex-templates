\begin{exercise}
	Prove that a taco is a hot dog.
\end{exercise}

\begin{proof}
	Let $\tau$ be a taco. By Lemma 69 from the text, we can choose
	a hamburger $H$ such that $\|H-\tau\|_{\text{hot dog}}<1/n$
	for a positive integer $n$.
\end{proof}
